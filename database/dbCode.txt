CREATE TABLE `raccoonsyze`.`usuarios_registrados` (`nombres` VARCHAR(55) NOT NULL , `apellidos` VARCHAR(55) NOT NULL , `password` VARCHAR(255) NOT NULL , `email` VARCHAR(255) NOT NULL , `telefono` INT(12) NOT NULL , `asunto` VARCHAR(100) NOT NULL , `necesidad` TEXT NOT NULL , `id` INT NOT NULL AUTO_INCREMENT , PRIMARY KEY (`id`), UNIQUE (`telefono`), UNIQUE (`email`)) ENGINE = InnoDB;
